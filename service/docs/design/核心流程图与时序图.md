## 1. 认证流程时序图

### 1.1 登录流程

```plantuml
@startuml
!theme plain

actor "客户端" as C
participant "AuthRouter" as R
participant "Depends\nget_auth_service" as D
participant "AuthService" as S
participant "PasswordService" as PS
participant "TokenService" as TS
participant "UserRepository" as UR
participant "RoleRepository" as RR
participant "MenuRepository" as MR
participant "CacheService" as CS
participant "Database" as DB

== 登录请求 ==

C -> R: POST /api/system/login\n{username, password}

R -> D: 获取 AuthService 实例

D -> UR: 创建 UserRepository(session)
D -> RR: 创建 RoleRepository(session)
D -> MR: 创建 MenuRepository(session)
D -> TS: 创建 TokenService(settings)
D -> CS: 创建 CacheService(redis)
D -> S: 创建 AuthService(session, repos, services)

R -> S: login(LoginDTO)

S -> CS: get_user_info(username)
alt 缓存未命中
    CS -->> S: None
    
    S -> UR: get_by_username(username)
    UR -> DB: SELECT * FROM sys_users\nWHERE username=?
    DB -->> UR: User ORM 对象
    UR -> UR: to_domain() 转换
    UR -->> S: UserEntity 对象
    
    S -> CS: set_user_info(username, user)
end

S -> PS: verify_password(input, hash)
PS -->> S: True/False

alt 密码错误
    S -->> R: raise UnauthorizedError
    R -->> C: 401 {code: 401, message: "用户名或密码错误"}
else 密码正确
    S -> TS: create_access_token(user_id, username)
    TS -->> S: access_token
    
    S -> TS: create_refresh_token(user_id, username)
    TS -->> S: refresh_token
    
    S -> CS: get_user_menus(user_id)
    alt 缓存未命中
        CS -->> S: None
        
        S -> RR: get_user_roles(user_id)
        RR -> DB: JOIN sys_userinfo_roles + sys_roles
        DB -->> RR: RoleEntity 列表
        RR -->> S: roles
        
        S -> MR: get_user_all_menus(user_id)
        MR -> DB: 单次 JOIN 查询\n用户→角色→菜单
        DB -->> MR: MenuEntity 列表
        MR -->> S: menus
        
        S -> CS: set_user_menus(user_id, menus)
    end
    
    S -->> R: {accessToken, refreshToken, roles, permissions}
    R -->> C: 200 {code: 0, data: {...}}
end

@enduml
```

### 1.2 Token 刷新流程

```plantuml
@startuml
!theme plain

actor "客户端" as C
participant "AuthRouter" as R
participant "AuthService" as S
participant "TokenService" as TS
participant "UserRepository" as UR
participant "RoleRepository" as RR
participant "MenuRepository" as MR

C -> R: POST /api/system/refresh-token\n{refreshToken}
R -> S: refresh_token(refresh_token)

S -> TS: decode_token(refresh_token)
TS -->> S: payload

S -> TS: verify_token_type(payload, "refresh")
alt token 类型不匹配
    S -->> C: 401 {message: "无效的刷新令牌"}
else 类型正确
    S -> UR: get_by_id(user_id)
    UR -->> S: user
    
    alt 用户不存在或不活跃
        S -->> C: 401 {message: "用户不可用"}
    else 用户有效
        S -> TS: create_access_token(user_id, username)
        TS -->> S: new_access_token
        
        S -> TS: create_refresh_token(user_id, username)
        TS -->> S: new_refresh_token
        
        S -> RR: get_user_roles(user_id)
        S -> MR: get_role_menu_ids(role_ids)
        
        S -->> C: 200 {code: 0, data: {accessToken, refreshToken, ...}}
    end
end

@enduml
```

## 2. RBAC 权限校验时序图

```plantuml
@startuml
!theme plain

actor "客户端" as C
participant "路由端点" as R
participant "require_permission" as P
participant "get_current_user_id" as AUTH
participant "TokenService" as TS
participant "CacheService" as CS
participant "UserRepository" as UR
participant "MenuRepository" as MR
participant "Database" as DB

C -> R: DELETE /api/system/user/{id}\nAuthorization: Bearer <token>

R -> P: Depends(require_permission)
P -> AUTH: Depends(get_current_user_id)

AUTH -> TS: decode_token(access_token)
TS -->> AUTH: payload {sub: user_id, type: "access"}
AUTH -> TS: verify_token_type(payload, "access")

AUTH -> CS: is_token_blacklisted(jti)
CS -->> AUTH: False (未在黑名单)

AUTH -->> P: user_id

P -> CS: get_user_info(user_id)
alt 缓存命中
    CS -->> P: UserEntity
else 缓存未命中
    P -> UR: get_by_id(user_id)
    UR -> DB: SELECT FROM sys_users
    DB -->> UR: User ORM
    UR -> UR: to_domain()
    UR -->> P: UserEntity
    P -> CS: set_user_info(user_id, user)
end

alt is_superuser == 1
    P -->> R: 超级用户，跳过权限检查 ✅
else 普通用户
    P -> CS: get_user_menus(user_id)
    alt 缓存命中
        CS -->> P: menus
    else 缓存未命中
        P -> MR: get_user_all_menus(user_id)
        MR -> DB: 单次 JOIN 查询\n用户→角色→菜单
        DB -->> MR: MenuEntity 列表
        MR -->> P: menus
        P -> CS: set_user_menus(user_id, menus)
    end
    
    alt menus 中存在 name=="user:delete" 且 menu_type==2
        P -->> R: 权限校验通过 ✅
    else 无对应权限
        P -->> C: 403 {code: 403, message: "权限不足"}
    end
end

R -> R: 执行业务逻辑
R -->> C: 200 {code: 0, data: ...}

@enduml
```

## 4. 请求处理全流程图

```plantuml
@startuml
!theme plain

|客户端请求| as A

if () then (CORS 检查)
    if () then (不通过)
        :[403 CORS 错误]
    else (通过)
        :[RequestLoggingMiddleware\n记录请求日志]
        
        if () then (需要认证?)
            :[直接进入路由处理]
        else (受保护接口)
            :[get_current_user_id\n解码 JWT Token]
            
            if () then (Token 有效?)
                if () then (在黑名单?)
                    :[401 Token 已注销]
                else (不在黑名单)
                    if () then (需要权限?)
                        :[require_permission\n缓存优先 + 单次 JOIN 查询]
                        if () then (权限足够?)
                            :[权限校验通过]
                        else (权限不足)
                            :[403 Forbidden]
                        end
                    end
                end
            else (无效/过期)
                :[401 Unauthorized]
            end
        end
    end
end

: [调用应用服务]

if () then (操作成功)
    :[get_db 自动 commit]
    :[统一响应 success_response]
else (操作失败)
    :[get_db 自动 rollback]
    
    if () then (AppError)
        :[ExceptionHandler\n返回业务错误]
    else if () then (ValidationError)
        :[422 参数校验错误]
    else
        :[500 内部错误]
    end
end

:[返回响应给客户端]

note right
非 GET 请求: 写入操作审计日志\nsys_operation_logs
end note

@enduml
```

## 6. 依赖注入装配流程图

```plantuml
@startuml
!theme plain

|HTTP 请求| as REQ

box "Depends 链"
    participant "get_db" as GET_DB
    participant "get_xxx_service" as GET_SVC
end box

box "服务实例"
    participant "XxxService" as SVC
end box

box "权限校验"
    participant "require_permission" as PERM
    participant "get_current_user_id" as CURR_USER
    participant "CacheService" as CACHE
    participant "MenuRepository" as MENU_REPO
end box

REQ -> GET_DB: Depends get_db
GET_DB -> SESSION: AsyncSession\nauto commit/rollback

REQ -> GET_SVC: Depends get_xxx_service

GET_SVC -> TOKEN_SVC: Depends get_token_service\nnew TokenService settings
GET_SVC -> PWD_SVC: Depends get_password_service\nPasswordService 静态
GET_SVC -> CACHE_SVC: Depends get_cache_service\nCacheService redis
GET_SVC -> REPO: Depends get_xxx_repository\nnew XxxRepository session

REPO -> SESSION

GET_SVC -> SVC: 注入 session + repos + domain_services + cache

REQ -> PERM: Depends require_permission

PERM -> CURR_USER: Depends get_current_user_id\nTokenService.decode + 黑名单检查
CURR_USER -> CACHE: 优先缓存查询
CURR_USER -> MENU_REPO: get_user_all_menus\n单次 JOIN 查询

CACHE -> SESSION
MENU_REPO -> SESSION

SVC -> EXEC: 执行业务逻辑
PERM -> EXEC

@enduml
```

## 7. 异常处理流程图

```plantuml
@startuml
!theme plain

|请求处理| as REQ

if () then (结果)
    :[统一响应 success_response\ncode: 0]
else if () then (AppError 子类)
    if () then (NotFoundError)
        :[404 code: 404]
    else if () then (ConflictError)
        :[409 code: 409]
    else if () then (UnauthorizedError)
        :[401 code: 401]
    else if () then (ForbiddenError)
        :[403 code: 403]
    else if () then (ValidationError)
        :[422 code: 422]
    else if () then (RateLimitError)
        :[429 code: 429]
    else
        :[400 code: 400]
    end
else if () then (RequestValidationError)
    :[422 参数校验失败\ndetail: 字段错误列表]
else
    :[500 内部服务器错误\ncode: 500]
end

: [统一响应格式\ncode + message + data]

@enduml
```

## 8. 仓储层 CRUD 时序图

```plantuml
@startuml
!theme plain

actor "Service" as S
participant "GenericRepository" as GR
participant "AsyncSession" as SESS
participant "Database" as DB

== 查询操作 (get_by_id) ==

S -> GR: get_by_id(id)
GR -> SESS: session.get(Model, id)
SESS -> DB: SELECT FROM table WHERE id=?
DB -->> SESS: Model ORM
SESS -->> GR: Model
GR -> GR: model.to_domain()
GR -->> S: Entity

== 查询操作 (get_one_by) ==

S -> GR: get_one_by(field, value)
GR -> GR: build select statement
GR -> SESS: session.exec(stmt)
SESS -> DB: SELECT FROM table WHERE field=?
DB -->> SESS: Model
SESS -->> GR: Model
GR -> GR: model.to_domain()
GR -->> S: Entity | None

== 创建操作 (create) ==

S -> GR: create(entity)
GR -> GR: model = Model.from_domain(entity)
GR -> SESS: session.add(model)
GR -> SESS: session.flush()
GR -> SESS: session.refresh(model)
GR -> GR: model.to_domain()
GR -->> S: Entity (with id)

== 更新操作 (update) ==

S -> GR: update(entity)
GR -> SESS: session.merge(entity)
SESS -> GR: merged Model
GR -> SESS: session.flush()
GR -> SESS: session.refresh(merged)
GR -> GR: merged.to_domain()
GR -->> S: Entity

== 删除操作 (delete) ==

S -> GR: delete(id)
GR -> GR: stmt = delete(Model).where(Model.id == id)
GR -> SESS: session.exec(stmt)
SESS -> DB: DELETE FROM table WHERE id=?
DB -->> SESS: rowcount
SESS -->> GR: rowcount
GR -->> S: bool (rowcount > 0)

== 批量删除 (batch_delete) ==

S -> GR: batch_delete(ids)
GR -> GR: stmt = delete(Model).where(Model.id.in_(ids))
GR -> SESS: session.exec(stmt)
SESS -> DB: DELETE FROM table WHERE id IN (...)
DB -->> SESS: rowcount
SESS -->> GR: rowcount
GR -->> S: int (deleted count)

@enduml
```
